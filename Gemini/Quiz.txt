### КОД (GameController.cs)

Запоминай логику: **Массивы, Стек, Простая математика.**

```csharp
using UnityEngine;
using UnityEngine.UI;
using UnityEngine.SceneManagement;
using System.Collections.Generic; // Нужно для List и Stack

public class GameController : MonoBehaviour
{
    public string word = "UNITY";    // Слово
    public Image img;                // Картинка (Fill)
    public Text resultText;          // Текст ввода
    public Button[] letterBtns;      // Массив кнопок с буквами

    Stack<Button> history = new Stack<Button>(); // История нажатий
    int errors = 0;

    void Start()
    {
        img.fillAmount = 0.25f; // Старт с 25%
        SetupButtons();
    }

    void SetupButtons()
    {
        // 1. Собираем буквы: само слово + случайные
        List<char> chars = new List<char>(word);
        while (chars.Count < letterBtns.Length) 
            chars.Add((char)Random.Range('A', 'Z' + 1));

        // 2. Мешаем (однострочный шаффл)
        chars.Sort((a, b) => Random.value > 0.5f ? 1 : -1);

        // 3. Назначаем на кнопки
        for (int i = 0; i < letterBtns.Length; i++)
        {
            Button btn = letterBtns[i];
            string letter = chars[i].ToString();
            
            btn.GetComponentInChildren<Text>().text = letter;
            btn.onClick.AddListener(() => TypeLetter(letter, btn)); // Лямбда
        }
    }

    void TypeLetter(string letter, Button btn)
    {
        resultText.text += letter;
        btn.interactable = false; // Отключаем кнопку
        history.Push(btn);        // Запоминаем кнопку
    }

    public void OnBackspace()
    {
        if (history.Count > 0)
        {
            resultText.text = resultText.text.Substring(0, resultText.text.Length - 1);
            history.Pop().interactable = true; // Возвращаем кнопку к жизни
        }
    }

    public void OnCheck()
    {
        if (resultText.text == word)
        {
            img.fillAmount = 1; // Полностью открыть
            Invoke("Win", 4);   // Ждать 4 сек
        }
        else
        {
            errors++;
            // Логика: 0 ош=0.25, 1 ош=0.5, 2 ош=0.75, 3 ош=1.0
            img.fillAmount = 0.25f + (errors * 0.25f);
        }
    }

    void Win() => SceneManager.LoadScene("Menu");
}
```

### Куда что привязать (Drag & Drop):

1.  Создай на сцене пустой объект **GameManager** и кинь на него этот скрипт.
2.  **Img:** Перетащи туда твою картинку (Image).
3.  **Result Text:** Перетащи туда текстовое поле, куда пишутся буквы.
4.  **Letter Btns:**
    *   В инспекторе скрипта найди `Letter Btns`.
    *   Поставь `Size` (например, 10 или 12, сколько у тебя кнопок).
    *   Перетащи все кнопки с буквами из иерархии в эти слоты.
5.  **Кнопки управления (в инспекторе самих кнопок `On Click`):**
    *   Кнопка **Стереть (Back)**: перетащи GameManager -> выбери функцию `OnBackspace()`.
    *   Кнопка **Проверить (Check)**: перетащи GameManager -> выбери функцию `OnCheck()`.

### Советы для чемпионата:
*   **Случайная буква:** `(char)Random.Range('A', 'Z' + 1)` — это проще запомнить, чем писать массив алфавита вручную.
*   **Сортировка:** `chars.Sort(...)` — запомни эту строку, она заменяет 10 строк сложного алгоритма перемешивания.
*   **Invoke:** Используй `Invoke("Win", 4)`, чтобы не писать сложные корутины (`IEnumerator`) таймеров.
